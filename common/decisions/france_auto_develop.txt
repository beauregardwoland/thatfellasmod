		}

		}

		is_good = yes
	
	
		days_mission_timeout = fra_evacuate_eco_timer

		activation = {
			always = no
		}

			hidden_effect = {
				every_core_state = {
					remove_dynamic_modifier = { modifier = GER_break_france_1 }

			}
		}
		timeout_effect = { 
			if = {
				limit = {
					check_variable = { fra_evacuate_eco_timer > 40 }
				}
				add_to_variable = { fra_evacuate_eco_timer = -14 }
				activate_mission = fra_evacuate_eco
				16 = {
					remove_building = {
						type = industrial_complex
						level = 4

						type = arms_factory

						type = industrial_complex

						level = 2

					}

					level = 2
				}

				limit = { any_core_state = { has_dynamic_modifier = GER_break_france_6 } }
				add_cic = 500
			}
			else = {
				if = {
						limit = { any_core_state = { has_dynamic_modifier = GER_break_france_5 } }
						every_core_state = {
							add_dynamic_modifier = { modifier = GER_break_france_6 days = 4000 scope = GER }
							remove_dynamic_modifier = { modifier = GER_break_france_5 }
						}
					}
					else_if = {
						limit = { any_core_state = { has_dynamic_modifier = GER_break_france_4 } }
						every_core_state = {
							add_dynamic_modifier = { modifier = GER_break_france_5 days = 4000 scope = GER }
							remove_dynamic_modifier = { modifier = GER_break_france_4
						}
					}
					else_if = {
						limit = { any_core_state = { has_dynamic_modifier = GER_break_france_3 } }
						every_core_state = {
							add_dynamic_modifier = { modifier = GER_break_france_4 days = 4000 scope = GER }
							remove_dynamic_modifier = { modifier = GER_break_france_3 }
						}
					}
					else_if = {
						limit = { any_core_state = { has_dynamic_modifier = GER_break_france_2 } }

							add_dynamic_modifier = { modifier = GER_break_france_3 days = 4000 scope = GER }
							remove_dynamic_modifier = { modifier = GER_break_france_2 }
						}
					}
					else_if = {
						limit = { any_core_state = { has_dynamic_modifier = GER_break_france_1 } }
						every_core_state = {
							add_dynamic_modifier = { modifier = GER_break_france_2 days = 4000 scope = GER }
							remove_dynamic_modifier = { modifier = GER_break_france_1 }
						}
					}
					else = {
						every_core_state = {
							add_dynamic_modifier = { modifier = GER_break_france_1 days = 4000 scope = GER }
						}
					}
				activate_mission = fra_evacuate_eco
					16 = {
						remove_building = {
							type = industrial_complex
							level = 2
						}
						remove_building = {
							type = arms_factory
							level = 2
						}
						add_extra_state_shared_building_slots = -4
					}
					779 = {
						add_extra_state_shared_building_slots = 1
						add_building_construction = {
							type = industrial_complex
							level = 1
							instant_build = yes
						}

							type = arms_factory
							level = 1

					}

						type = industrial_complex
						level = 1
					}
					add_offsite_building = {
						type = arms_factory
						level = 1
					}
			}
