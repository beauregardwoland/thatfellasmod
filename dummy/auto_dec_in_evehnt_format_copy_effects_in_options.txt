add_namespace = auto_dec_wars  ###copy and paste effect in option to use in other events

country_event = {  #####german declaration of war on belgium
	id = auto_dec_wars.1
	title = auto_dec_wars.1.t
	desc = auto_dec_wars.1.d
	picture = GFX_report_event_chinese_soldiers_fighting
	
	is_triggered_only = yes
	
	option = { 
		name = 	auto_dec_wars.1.a
		ai_chance = { factor = 100 }
		every_country = {  ###all central powers members dec on belgium
			limit = {
				is_in_faction_with = GER
			}
			declare_war_on = {
				target = BEL
				type = annex_everything
			}
		}
		GER = { ###germany dec on belgium
			declare_war_on = {
				target = BEL
				type = annex_everything
			}
		}
	}
}

					target = PREV
					type = annex_everything
				}
			}

			limit = {
				or = { is_subject_of = ENG tag = ENG  }  ####scopes to england and dominions

				declare_war_on = {
					target = PREV
					type = annex_everything
				}
			}
		}

	}
}
